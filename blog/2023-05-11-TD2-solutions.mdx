## Exercise 1:

Here's an example of how you can achieve the desired layout using React Native and Expo:

```javascript
import React from 'react';
import { View, Text, StyleSheet } from 'react-native';

const Exercise1 = () => {
  return (
    <View style={styles.container}>
      <Text style={styles.header}>Here are my boxes</Text>
      <View style={styles.boxContainer}>
        <Rectangle color="#2aa198" name="Cyan: #2aa198" />
        <Rectangle color="#268bd2" name="Blue: #268bd2" />
        <Rectangle color="#d33682" name="Magenta: #d33682" />
        <Rectangle color="#cb4b16" name="Orange: #cb4b16" />
      </View>
    </View>
  );
};

const Rectangle = ({ color, name }) => {
  return (
    <View style={[styles.rectangle, { backgroundColor: color }]}>
      <Text style={styles.rectangleText}>{name}</Text>
    </View>
  );
};

const styles = StyleSheet.create({
  container: {
    flex: 1,
    padding: 16,
    alignItems: 'center',
  },
  header: {
    fontSize: 24,
    fontWeight: 'bold',
    marginBottom: 16,
  },
  boxContainer: {
    justifyContent: 'center',
  },
  rectangle: {
    width: 200,
    height: 50,
    marginBottom: 8,
    justifyContent: 'center',
    alignItems: 'center',
  },
  rectangleText: {
    color: 'white',
    fontWeight: 'bold',
  },
});

export default Exercise1;
```

You can use the `Exercise1` component in your main app file to display the header and rectangles with their respective colors and names.

## Exercise 4:

Here's an example of how you can create a contact list using the FlatList component:

```javascript
import React from 'react';
import { View, Text, FlatList, StyleSheet } from 'react-native';

const ContactScreen = () => {
  const contacts = [
    { id: '1', name: 'John Doe', phoneNumber: '123-456-7890' },
    { id: '2', name: 'Jane Smith', phoneNumber: '987-654-3210' },
    { id: '3', name: 'Alice Johnson', phoneNumber: '555-123-4567' },
  ];

  const renderContactItem = ({ item }) => (
    <View style={styles.item}>
      <Text style={styles.name}>{item.name}</Text>
      <Text style={styles.phoneNumber}>{item.phoneNumber}</Text>
    </View>
  );

  return (
    <View style={styles.container}>
      <FlatList
        data={contacts}
        renderItem={renderContactItem}
        keyExtractor={(item) => item.id}
        ItemSeparatorComponent={() => <View style={styles.separator} />}
      />
    </View>
  );
};

const styles = StyleSheet.create({
  container: {
    flex: 1,
    padding: 16,
  },
  item: {
    paddingVertical: 8,
  },
  name: {
    fontSize: 16,
    fontWeight: 'bold',
  },
  phoneNumber: {
    fontSize: 14,
    color: '#666',
  },
  separator: {
    height: 1,
    backgroundColor: '#ccc',
  },
});

export default ContactScreen;
```

You can use the `ContactScreen` component in your main app file to display the contact list using the FlatList component.

## Exercise 5:

Here's an example of how you can group contacts by the first letter of their names using the SectionList component:

```javascript
import React from 'react';
import { View, Text, SectionList, StyleSheet } from 'react-native';

const SectionContactScreen = () => {
  const contacts = [
    { id: '1', name: 'John Doe', phoneNumber: '123-456-7890' },
    { id: '2', name: 'Jane Smith', phoneNumber: '987-654-3210' },
   

 { id: '3', name: 'Alice Johnson', phoneNumber: '555-123-4567' },
    // ...more contacts
  ];

  const contactsBySection = contacts.reduce((sections, contact) => {
    const firstLetter = contact.name.charAt(0).toUpperCase();
    if (!sections[firstLetter]) {
      sections[firstLetter] = { title: firstLetter, data: [] };
    }
    sections[firstLetter].data.push(contact);
    return sections;
  }, {});

  const sections = Object.values(contactsBySection);

  const renderContactItem = ({ item }) => (
    <View style={styles.item}>
      <Text style={styles.name}>{item.name}</Text>
      <Text style={styles.phoneNumber}>{item.phoneNumber}</Text>
    </View>
  );

  const renderSectionHeader = ({ section }) => (
    <Text style={styles.sectionHeader}>{section.title}</Text>
  );

  return (
    <View style={styles.container}>
      <SectionList
        sections={sections}
        renderItem={renderContactItem}
        renderSectionHeader={renderSectionHeader}
        keyExtractor={(item) => item.id}
        ItemSeparatorComponent={() => <View style={styles.separator} />}
      />
    </View>
  );
};

const styles = StyleSheet.create({
  container: {
    flex: 1,
    padding: 16,
  },
  item: {
    paddingVertical: 8,
  },
  name: {
    fontSize: 16,
    fontWeight: 'bold',
  },
  phoneNumber: {
    fontSize: 14,
    color: '#666',
  },
  sectionHeader: {
    fontSize: 18,
    fontWeight: 'bold',
    backgroundColor: '#f5f5f5',
    paddingVertical: 8,
    paddingHorizontal: 16,
  },
  separator: {
    height: 1,
    backgroundColor: '#ccc',
  },
});

export default SectionContactScreen;
```

You can use the `SectionContactScreen` component in your main app file to display the contact list grouped by the first letter of their names using the SectionList component.
